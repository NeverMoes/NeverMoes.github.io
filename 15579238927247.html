示例

常用注解：

  1. 测试方法注解：
    1. @Test，标注测试方法，测试阶段会运行所有被注解的方法。
    2. @ParameterizedTest，同 @Test 但是可以使其以不同的参数运行多次。
    3. @RepeatedTest，重复测试。
    4. @TestFactory，标注动态测试。
  2. @BeforeEach/@AfterEach，在每一个测试方法运行前/后都运行一个指定的方法。
  3. @BeforeAll/@AfterAll，在整个测试类运行前/后都运行一个指定的方法。
  4. @Disabled，指定不再运行。

  5. @Nested，添加嵌套层级。

  6. @DisplayName，测试的显示名称。

  7. @Tag，添加 Tag，可以有选择性地执行。

class StandardTests {

    @BeforeAll
    static void initAll() {
    }

    @BeforeEach
    void init() {
    }

    @Test
    void Test() {
    }
    
    @Nest
    @DisplayName("这是一个测试")
    class ATest() {
        @Test
        void nestedTest() {
        }
    }

    @Test
    @Disabled("for demonstration purposes")
    void skippedTest() {
        // not executed
    }

    @AfterEach
    void tearDown() {
    }

    @AfterAll
    static void tearDownAll() {
    }
}

Assert

class AssertionsDemo {

    private final Calculator calculator = new Calculator();
    private final Person person = new Person("Jane", "Doe");

    @Test
    void standardAssertions() {
        assertEquals(4, calculator.multiply(2, 2), "可选的失败信息");

        // 组断言
        assertAll("person",
            () -> assertEquals("Jane", person.getFirstName()),
            () -> assertEquals("Doe", person.getLastName())
        );

        // 断言依赖，只要其中一个断言失败了，就不会继续执行
        assertAll("properties",
            () -> {
                String firstName = person.getFirstName();
                assertNotNull(firstName);
                // 只有上一个断言成功才会执行
                assertAll("first name",
                    () -> assertTrue(firstName.startsWith("J")),
                    () -> assertTrue(firstName.endsWith("e"))
                );
            }
        );
        
        // 对象相等
        assertSame(person, Person)
        
        // 迭代器全部元素相等
        assertIterableEquals
        
        // 数组元素相等
        assertArrayEquals
        
        // 异常断言
        Exception exception = assertThrows(ArithmeticException.class, () ->
            calculator.divide(1, 0));
        assertEquals("/ by zero", exception.getMessage());

        // 超时断言
        assertTimeout(ofMinutes(2), () -> {
            Thread.sleep();
        });

        // 超时断言正常返回
        String actualResult = assertTimeout(ofMinutes(2), () -> {
            return "a result";
        });
        assertEquals("a result", actualResult);

        // 如果没有超时则调用方法
        String actualGreeting = assertTimeout(ofMinutes(2), AssertionsDemo::greeting);
        assertEquals("Hello, World!", actualGreeting);
        
        // 超时断言，区别在于测试程序是否在同一线程内执行
        assertTimeoutPreemptively
    }

    private static String greeting() {
        return "Hello, World!";
    }
}

Hamcrest

Hamcrest 是 JUnit 官方推荐的第三方断言框架。
首先要理解两个概念：

  1. MatcherAssert，断言匹配器，将一个执行结果和一个匹配对象进行匹配。
  2. Matchers，匹配对象。

详细可以参考：

  1. The Javadoc of the MatcherAssert class
  2. The Javadoc of the Matchers class
import org.junit.jupiter.api.DisplayName;
import org.junit.jupiter.api.Nested;
import org.junit.jupiter.api.Test;
 
import static org.hamcrest.MatcherAssert.*;
import static org.hamcrest.Matchers.*;
 
class ObjectAssertionTest {
    @Test
    void testExample() {
        // 匹配器与匹配对象
        assertThat(ACTUAL, is(EXPECTED));
        assertThat(ACTUAL, not(EXPECTED));
        assertThat(true, is(true));
        
        // null 值
        assertThat(null, nullValue());
        assertThat(new Object(), notNullValue());
        
        // 实例
        assertThat(ACTUAL, sameInstance(EXPECTED));
        assertThat(ACTUAL, not(sameInstance(EXPECTED)));
        
        // list 相关
        // 匹配对象可以断言两个数组
        assertThat(array, is(EXPECTED));
        // 
        assertThat(list, hasSize(2));
        // 有序，first 必须为 list 的第一个元素
        assertThat(list, contains(first, second));
        // 无序
        assertThat(list, containsInAnyOrder(second, first));
        assertThat(list, hasItem(first));
        // not 同理
        assertThat(list, not(hasItem(new Object())));
        
        // map 相关
        assertThat(map, hasKey(KEY));
        assertThat(map, hasEntry(KEY, VALUE));
        
        // 对象
        // 属性断言
        assertThat(person, allOf(
        hasProperty("firstName", is(FIRST_NAME)),
        hasProperty("lastName", is(LAST_NAME))
        ));
        // 全部匹配与任意匹配
        assertThat(person, anyOf(
        hasProperty("firstName", is(FIRST_NAME)),
        hasProperty("lastName", is(LAST_NAME))
        ));
        
        // 自定义提示信息
        assertThat(String.format(
                "The list doesn't contain the expected object: %s", 
                first
            ),
            list,
            hasItem(first)
        );
    }
}

AssertJ

AssertJ 是另一个断言库，主要支持了流式断言。

// Junit 中的断言
String s = "abcde";
Assertions.assertTrue(s.startsWith("ab")) ;
Assertions.assertTrue(s.endsWith ("ed"));
Assertions.assertEquals(s,s.length()) ;

// AssertJ 中的流式断言
Assertions.assertThat(s).as("字符串判断：判断首尾长度")
    .startsWith("ab").endsWith("de").hasSize (5);

Assumption

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assumptions.assumeTrue;
import static org.junit.jupiter.api.Assumptions.assumingThat;

import example.util.Calculator;
import org.junit.jupiter.api.Test;

class AssumptionsDemo {

    private final Calculator calculator = new Calculator();

    @Test
    void testOnlyOnCiServer() {
        assumeTrue("CI".equals(System.getenv("ENV")));
        // remainder of test
    }

    @Test
    void testOnlyOnDeveloperWorkstation() {
        assumeTrue("DEV".equals(System.getenv("ENV")),
            () -> "Aborting test: not on developer workstation");
        // remainder of test
    }

    @Test
    void testInAllEnvironments() {
        assumingThat("CI".equals(System.getenv("ENV")),
            () -> {
                // perform these assertions only on the CI server
                assertEquals(2, calculator.divide(4, 2));
            });
        // perform these assertions in all environments
        assertEquals(42, calculator.multiply(6, 7));
    }
}

条件测试

/*
注意注解是可以作为元注解来嵌套使用的，比如 MAC 的两个例子。
*/

@Test
@EnabledOnOs(MAC)
void onlyOnMacOs() {
    // ...
}

@TestOnMac
void testOnMac() {
    // ...
}

@Test
@EnabledOnOs({ LINUX, MAC })
void onLinuxOrMac() {
    // ...
}

@Test
@DisabledOnOs(WINDOWS)
void notOnWindows() {
    // ...
}

// 结合元注解使用
@Target(ElementType.METHOD)
@Retention(RetentionPolicy.RUNTIME)
@Test
@EnabledOnOs(MAC)
@interface TestOnMac {
}

// 运行时条件
@Test
@EnabledOnJre(JAVA_8)
void onlyOnJava8() {
    // ...
}

// 读取 JVM 配置作为条件
@Test
@EnabledIfSystemProperty(named = "os.arch", matches = ".*64.*")
void onlyOn64BitArchitectures() {
    // ...
}

// 读取环境变量
@Test
@EnabledIfEnvironmentVariable(named = "ENV", matches = "staging-server")
void onlyOnStagingServer() {
    // ...
}

// 甚至可以使用表达式的结果作为条件
@Test // Static JavaScript expression.
@EnabledIf("2 * 3 == 6")
void willBeExecuted() {
    // ...
}

@RepeatedTest(10) // Dynamic JavaScript expression.
@DisabledIf("Math.random() < 0.314159")
void mightNotBeExecuted() {
    // ...
}

// 从内置的字典中读取
@Test // Regular expression testing bound system property.
@DisabledIf("/32/.test(systemProperty.get('os.arch'))")
void disabledOn32BitArchitectures() {
    assertFalse(System.getProperty("os.arch").contains("32"));
}

// 一个更复杂的例子
@Test // Multi-line script, custom engine name and custom reason.
@EnabledIf(value = {
                "load('nashorn:mozilla_compat.js')",
                "importPackage(java.time)",
                "",
                "var today = LocalDate.now()",
                "var tomorrow = today.plusDays(1)",
                "tomorrow.isAfter(today)"
            },
            engine = "nashorn",
            reason = "Self-fulfilling: {result}")
void theDayAfterTomorrow() {
    LocalDate today = LocalDate.now();
    LocalDate tomorrow = today.plusDays(1);
    assertTrue(tomorrow.isAfter(today));
}
