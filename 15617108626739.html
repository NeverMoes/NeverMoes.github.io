模板

这里以对 [1,2,3] 的排列说明一下搜索的模板。
首先问题的解集是一棵树，但是这一类问题并不一定都是树，也有可能是图。
那么直接使用 dfs 搜索，dfs 模板如下：

/**
    resutl 总解集
    temp 暂存集，保存搜索到当前位置的解
    problem 问题集
    curent 用于标识搜索的位置
*/
dfs(result, temp, problem, curent) {
    // 先判断当前集合是否搜索到了终点
    if (temp) {
        // 如果暂存集满足条件则存入总解集合
        result.add(temp)
        // 搜索到终节点，终止
        return;
    }
    // 满足前进条件，即可以往树的下一层搜索的条件
    while(forward) {
        // 如果明显满足某些减枝条件就不进入下一层的搜索了
        if (cutBranch) {
            continue;
        }
        // 进入下一层搜索
        temp.add();
        dfs();
        // 回退
        temp.remove();
    }
}

        return resList;
    }
 
    public void dfs(List<List<Integer>> resList, List<Integer> tmpList, int[] nums){
        if (tmpList.size() == nums.length) {
            resList.add(new ArrayList<>(tmpList));
            return;
        }
        for (int i = 0; i < nums.length; i++) {
            // 无效解要跳过
            if (tmpList.contains(nums[i]))
                continue;
            tmpList.add(nums[i]);
            bt(resList, tmpList, nums);
            // 回退
            tmpList.remove(tmpList.size()-1);
        }
    }
}

思路：

复杂度：
时间：
空间：

public class Solution {
    public List<List<Integer>> permuteUnique(int[] nums) {
        List<List<Integer>> resList = new ArrayList<List<Integer>>();
        boolean[] used = new boolean[nums.length];
        // 这里排序一下主要是为了方便判断是否有相同的数字进入搜索了
        Arrays.sort(nums);
        dfs(nums, resList, new ArrayList<Integer>(), used);
        return resList;
    }

    public void dfs(int[] nums, List<List<Integer>> resList, List<Integer> tmpList, boolean[] used) {
        if (tmpList.size() == nums.length){
            resList.add(new ArrayList<Integer>(tmpList));
            return;
        }
        
        for (int i = 0; i < nums.length; i++) {
            // 跳过已经使用过的和重复的数字
            if (used[i])
                continue;
            // 排序的原因
            if (i > 0 && nums[i-1] == nums[i] && !used[i-1])
                continue;
            
            used[i]=true;
            tmpList.add(nums[i]);
            dfs(nums, resList, tmpList, used);
            // 回退
            used[i]=false;
            tmpList.remove(tmpList.size()-1);
        }
    }
}

问题：77. Combinations
给定 n 长度的序列，选出长度为 k 的组合。

思路：

复杂度：
时间：
空间：

class Solution {
    public List<List<Integer>> combine(int n, int k) {
        List<List<Integer>> resList = new ArrayList<>();
        // s 表示搜索的起始数字
        dfs(resList, new ArrayList<Integer>(), 1, n, k);
        return resList;
    }
    
    public void dfs(List<List<Integer>> resList, List<Integer> tmpList, int s, int n, int k) {
        if (k == 0) {
            resList.add(new ArrayList<Integer>(tmpList));
            return;
        }
        for (int i = s; i <= n; i++) {
            tmpList.add(i);
            dfs(resList, tmpList, i+1, n, k-1);
            // 回退
            tmpList.remove(tmpList.size() - 1);
        }
    }
}

问题：39. Combination Sum
给定一个不重复的数字集合，选出和等于给定目标值的组合。

思路：

复杂度：
时间：
空间：

class Solution {
    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        Arrays.sort(candidates);
        List<List<Integer>> resList = new ArrayList<>();
        dfs(resList, new ArrayList<Integer>(), candidates, target, 0, 0);
        return resList;
    }
    
    public void dfs(List<List<Integer>>resList, List<Integer> tmpList, int[] nums,
                   int target, int index, int curSum){
        if (curSum == target) {
            resList.add(new ArrayList<Integer>(tmpList));
            return;
        }

        else if (curSum > target) {
            return;
        }
        
        else{
            for(int i = index; i < nums.length; i++){
                tmpList.add(nums[i]);
                curSum += nums[i];
                dfs(resList, tmpList, nums, target, index, curSum);
                tmpList.remove(tmpList.size() - 1);
                curSum -= nums[i];
                index++;
            }
        }
    }
}

N 皇后

class Solution {
    public List<List<String>> solveNQueens(int n) {
        List<List<String>> resList = new ArrayList<>();
        // 二维棋盘 为 true 表示已经放了皇后
        boolean[][] board = new boolean[n][n];
        dfs(resList, board, n, 0);
        return resList;
    }

    public void dfs(List<List<String>> resList, boolean[][] board, int n, int curRow) {
        if (curRow == n) {
            resList.add(convert(board));
            return;
        }
        
        for (int col = 0; col < n; col++) {
            if (isValid(board, curRow, col, n)) {
                board[curRow][col] = true;
                dfs(resList, board, n, curRow + 1);
                board[curRow][col] = false;
            }
        }
    }
    
    public boolean isValid(boolean[][] board, int row, int col, int n) {
        // 同一列是否放了
        for (int i = 0; i < row; i++)
            if (board[i][col])
                return false;
        // 45度对角线
        for (int i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--)
            if (board[i][j])
                return false;
        // 135度对角线
        for (int i = row - 1, j = col + 1; i >= 0 && j < n; i--, j++)
            if (board[i][j])
                return false;
        // 不检查行是因为以行为标准进行的搜索
        return true;
    }
    
    public List<String> convert(boolean[][] board) {
        List<String> res = new ArrayList<>();
        int n = board.length;
        for (int i = 0; i < n; i++) {
            StringBuilder sb = new StringBuilder();
            for(int j = 0; j < n; j++) {
                sb.append(board[i][j] ? "Q" : ".");
            }
            res.add(sb.toString());
        }
        return res;
    }
}
